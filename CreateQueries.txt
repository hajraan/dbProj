SELECT * FROM demo;

CREATE TABLE PATIENT (
    patientId int NOT NULL,
    lname varchar not NULL,
    fname varchar not NULL,
  	address varchar not NULL,
  	phone varchar not NULL,
  	medicalHistory varchar,
  	HIPAALastSigned varchar,
  	xRayDate varchar,
  	payedLastYear int,
  	primary key (patientId)
);
CREATE TABLE PAT_PHONE (
     pId int,
    phone varchar,
  	FOREIGN key (pId) REFERENCES PATIENT(patientId)
);

CREATE TABLE PAT_MEDICAL_HISTORY (
     pId int,
    procedure varchar,
  	FOREIGN key (pId) REFERENCES PATIENT(patientId)
);

CREATE TABLE PAT_ALLERGIES (
     pId int,
    allergen varchar,
  FOREIGN key (pId) REFERENCES PATIENT(patientId)
);


CREATE TABLE OFFICE (
     officeId int not NULL,
    address varchar NOT NULL,
  primary key (officeId)
);

CREATE TABLE FACULTY (
     facultyId int not NULL,
    lname varchar not NULL,
    fname varchar not NULL,
  	workHours varchar not NULL,
  	salary int not NULL,
  	role varchar not NULL,
  	bday varchar not NULL,
  	sex varchar not NULL,
  	PRIMARY key (facultyId)
  
);

CREATE TABLE PAYMENT (
     paymentId int not NULL,
    patientId int,
    pType varchar not NULL,
  	amount int not NULL,
  primary key (paymentId),
  FOREIGN key (patientId) REFERENCES PATIENT(patientId)
  
);

CREATE TABLE CREDIT_CARD (
     cardNum varchar not NULL,
    CVV varchar not NULL,
    expiration varchar not NULL,
  primary key (cardNum)
);

CREATE TABLE CHEQUE (
    accountNo varchar,
    routingNo varchar,
  primary key (accountNo)
);

CREATE TABLE DEBIT (
     cardNum varchar,
    accountNo varchar,
    routingNo varchar,
  primary key (cardNum)
);

CREATE TABLE APPOINTMENT (
     appointmentId int not NULL,
    dateOf varchar not NULL,
    medicalProfessional varchar,
  	totalCost int not NULL,
  	faculty varchar,
  	patientId INT,
  FOREIGN key (patientId) REFERENCES PATIENT(patientId),
  PRIMARY key (appointmentId)
);

CREATE TABLE INSURANCE (
     memberId int,
    groupNum int,
    insuredId int,
  	holderName varchar,
  	phoneNumber varchar,
  	company varchar,
  	policy varchar,
  primary key (memberId),
  FOREIGN key (insuredId) REFERENCES PATIENT(patientId)
);

CREATE TABLE MEDICATION (
     medicationId int,
    patientId int,
    dose varchar,
  	name varchar,
  PRIMARY key (medicationId),
  FOREIGN key (patientId) REFERENCES PATIENT(patientId)
);

CREATE TABLE APPOINTMENT_PAYMENT (
     paymentId int,
    appointmentId int,
    amount int not NULL,
  	dateOf varchar not NULL,
  FOREIGN key (paymentId) REFERENCES PAYMENT(paymentId),
  FOREIGN key (appointmentId) REFERENCES APPOINTMENT(appointmentId)
);
CREATE TABLE SERVICE (
     serviceId int not NULL,
  	patientId int,
    amountDue int not NULL,
    dateDue varchar not NULL,
  	daysLate int not NULL,
  PRIMARY KEY (serviceId),
  FOREIGN key (patientId) REFERENCES PATIENT(patientId)
);

CREATE TABLE APPOINTMENT_SERVICE (
     serviceId int,
    appointmentId int
  FOREIGN key (serviceId) REFERENCES SERVICE(serviceId),
  FOREIGN key (appointmentId) REFERENCES APPOINTMENT(appointmentId)
);

CREATE TABLE SERVICE_PROCEDURES (
     serviceId int,
    appointmentId int,
   FOREIGN key (serviceId) REFERENCES SERVICE(serviceId),
  FOREIGN key (appointmentId) REFERENCES APPOINTMENT(appointmentId)
);

CREATE TABLE LICENSURE (
     facultyId int,
    expiryDate varchar not NULL,
    type varchar not NULL,
  FOREIGN key (facultyId) REFERENCES FACULTY(facultyId)
);

CREATE TABLE SUPPLY_SELLER (
     number int not NULL,
    location varchar not NULL,
    name varchar not NULL,
  PRIMARY KEY (number)
);

CREATE TABLE SUPPLY_PURCHASE (
     ownerName varchar not NULL,
    supplierNumber int not NULL,
    purchasePrice int not NULL,
  	dateOf varchar not NULL,
  	quantity int not NULL,
  PRIMARY key (supplierNumber)
);
CREATE TABLE PROCEDURE (
     type varchar not NULL,
    dateOf varchar not NULL,
    professionalName varchar not NULL,
  	cost int not NULL
);
